# 号码提取与通用信息提取综合方案

> **项目**：智能数据标注平台
> **目标**：支持准确提取身份证号、银行卡号、手机号，同时兼容通用信息提取
> **设计日期**：2025-12-30
> **版本**：V1.0

---

## 一、需求分析

### 1.1 核心需求

| 需求 | 描述 | 优先级 |
|------|------|--------|
| **身份证号提取** | 提取18位标准身份证号、15位旧版、不符合位数的疑似身份证号 | 高 |
| **银行卡号提取** | 提取16-19位银行卡号，与身份证号区分 | 高 |
| **手机号提取** | 提取11位手机号 | 高 |
| **通用信息提取** | 保持现有LLM通用提取能力，支持任意字段的自由提取 | 中 |

### 1.2 技术挑战

- **格式混淆**：16-19位数字可能是身份证号或银行卡号
- **上下文依赖**：需要结合语义理解准确分类
- **遮挡处理**：警情内容中可能有部分遮挡（如星号）
- **向后兼容**：必须兼容现有的通用提取功能

### 5.3 提取器实现

#### 5.3.1 身份证号提取器

```java
@Component
public class IdCardExtractor implements INumberExtractor {

    // 18位身份证号正则（严格模式）
    private static final Pattern ID_CARD_18_STRICT = Pattern.compile(
        "\\b(1[1-9]\\d{4})(19|20)\\d{2}(0[1-9]|1[0-2])(0[1-9]|[12]\\d|3[01])\\d{3}[0-9Xx]\\b"
    );

    // 15位身份证号正则（旧版）
    private static final Pattern ID_CARD_15 = Pattern.compile(
        "\\b[1-9]\\d{4}\\d{2}(0[1-9]|1[0-2])(0[1-9]|[12]\\d|3[01])\\d{3}\\b"
    );

    // 宽松模式（疑似身份证号）
    private static final Pattern ID_CARD_LOOSE = Pattern.compile(
        "\\b\\d{15,18}(?:\\*{1,3})?\\b"
    );

    @Override
    public List<ExtractedNumber> extract(String text, Map<String, Object> options) {
        List<ExtractedNumber> results = new ArrayList<>();

        boolean include18Digit = (boolean) options.getOrDefault("include18Digit", true);
        boolean include15Digit = (boolean) options.getOrDefault("include15Digit", true);
        boolean includeLoose = (boolean) options.getOrDefault("includeLoose", true);

        // 1. 提取18位标准身份证号
        if (include18Digit) {
            results.addAll(extract18DigitIdCards(text));
        }

        // 2. 提取15位旧版身份证号
        if (include15Digit) {
            results.addAll(extract15DigitIdCards(text));
        }

        // 3. 提取疑似身份证号
        if (includeLoose) {
            results.addAll(extractLooseIdCards(text, results));
        }

        return results;
    }

    private List<ExtractedNumber> extract18DigitIdCards(String text) {
        List<ExtractedNumber> results = new ArrayList<>();
        Matcher matcher = ID_CARD_18_STRICT.matcher(text);

        while (matcher.find()) {
            String idCard = matcher.group();
            if (validateCheckBit(idCard)) {
                results.add(ExtractedNumber.builder()
                    .value(idCard)
                    .confidence(0.95f)
                    .validation("18位标准格式，通过校验位验证")
                    .startIndex(matcher.start())
                    .endIndex(matcher.end())
                    .build());
            }
        }
        return results;
    }

    /**
     * 校验18位身份证号的校验位
     */
    private boolean validateCheckBit(String idCard) {
        int[] weights = {7, 9, 10, 5, 8, 4, 2, 1, 6, 3, 7, 9, 10, 5, 8, 4, 2};
        char[] checkChars = {'1', '0', 'X', '9', '8', '7', '6', '5', '4', '3', '2'};

        int sum = 0;
        for (int i = 0; i < 17; i++) {
            sum += (idCard.charAt(i) - '0') * weights[i];
        }

        char expectedCheck = checkChars[sum % 11];
        char actualCheck = Character.toUpperCase(idCard.charAt(17));

        return expectedCheck == actualCheck;
    }

    // ... 其他方法
}
```

#### 5.3.2 银行卡号提取器

```java
@Component
public class BankCardExtractor implements INumberExtractor {

    private static final Pattern BANK_CARD = Pattern.compile("\\b\\d{16,19}\\b");

    // 常见银行卡BIN码
    private static final Map<String, String> BANK_BIN_CODES = Map.ofEntries(
        Map.entry("622700", "中国工商银行"),
        Map.entry("622202", "中国工商银行"),
        Map.entry("622848", "中国农业银行"),
        Map.entry("621700", "中国建设银行"),
        Map.entry("521899", "招商银行")
        // ... 更多BIN码
    );

    @Override
    public List<ExtractedNumber> extract(String text, Map<String, Object> options) {
        List<ExtractedNumber> results = new ArrayList<>();
        Matcher matcher = BANK_CARD.matcher(text);

        while (matcher.find()) {
            String cardNumber = matcher.group();

            // Luhn算法校验
            boolean luhnValid = luhnCheck(cardNumber);

            // 识别发卡行
            String bank = "未知银行";
            if (cardNumber.length() >= 6) {
                String bin = cardNumber.substring(0, 6);
                bank = BANK_BIN_CODES.getOrDefault(bin, "未知银行");
            }

            results.add(ExtractedNumber.builder()
                .value(cardNumber)
                .confidence(luhnValid ? 0.9f : 0.7f)
                .validation(bank + (luhnValid ? "，通过Luhn校验" : "，未通过Luhn校验"))
                .startIndex(matcher.start())
                .endIndex(matcher.end())
                .build());
        }

        return results;
    }

    /**
     * Luhn算法校验
     */
    private boolean luhnCheck(String cardNumber) {
        int sum = 0;
        boolean alternate = false;

        for (int i = cardNumber.length() - 1; i >= 0; i--) {
            int digit = cardNumber.charAt(i) - '0';
            if (alternate) {
                digit *= 2;
                if (digit > 9) digit -= 9;
            }
            sum += digit;
            alternate = !alternate;
        }

        return sum % 10 == 0;
    }
}
```

#### 5.3.3 手机号提取器

```java
@Component
public class PhoneExtractor implements INumberExtractor {

    private static final Pattern PHONE_STRICT = Pattern.compile(
        "\\b(1[3-9])\\d{9}\\b"
    );

    // 运营商号段
    private static final Map<String, String> OPERATORS = Map.of(
        "130", "中国联通", "131", "中国联通", "132", "中国联通",
        "133", "中国电信", "153", "中国电信", "180", "中国电信",
        "134", "中国移动", "135", "中国移动", "136", "中国移动",
        "137", "中国移动", "138", "中国移动", "139", "中国移动"
        // ... 更多号段
    );

    @Override
    public List<ExtractedNumber> extract(String text, Map<String, Object> options) {
        List<ExtractedNumber> results = new ArrayList<>();
        Matcher matcher = PHONE_STRICT.matcher(text);

        while (matcher.find()) {
            String phone = matcher.group();

            // 识别运营商
            String operator = "未知运营商";
            if (phone.length() >= 3) {
                String prefix = phone.substring(0, 3);
                operator = OPERATORS.getOrDefault(prefix, "未知运营商");
            }

            results.add(ExtractedNumber.builder()
                .value(phone)
                .confidence(0.92f)
                .validation(operator)
                .startIndex(matcher.start())
                .endIndex(matcher.end())
                .build());
        }

        return results;
    }
}
```

### 5.4 提取器协调服务

```java
@Service
public class ExtractionOrchestrator {

    @Autowired
    private IdCardExtractor idCardExtractor;

    @Autowired
    private BankCardExtractor bankCardExtractor;

    @Autowired
    private PhoneExtractor phoneExtractor;

    @Autowired
    private DeepSeekService deepSeekService;

    /**
     * 统一提取入口
     * @param label 标签配置
     * @param rowData 数据行
     * @return 提取结果
     */
    public Map<String, Object> extract(Label label, Map<String, Object> rowData) {
        String type = label.getType();

        // 1. 结构化提取（新增）
        if (Label.Type.STRUCTURED_EXTRACTION.equals(type)) {
            return extractStructured(label, rowData);
        }

        // 2. LLM通用提取（已有）
        if (Label.Type.EXTRACTION.equals(type)) {
            if (label.getExtractFields() != null && !label.getExtractFields().isEmpty()) {
                return deepSeekService.extract(label, rowData, null);
            } else {
                return deepSeekService.extractFreeForm(label, rowData, null);
            }
        }

        throw new IllegalArgumentException("不支持的标签类型: " + type);
    }

    /**
     * 结构化提取
     */
    private Map<String, Object> extractStructured(Label label, Map<String, Object> rowData) {
        long startTime = System.currentTimeMillis();

        // 解析提取器配置
        JSONObject extractorConfig = parseExtractorConfig(label);

        String extractorType = extractorConfig.getString("extractorType");

        // 获取目标文本
        String text = getTargetText(label, rowData);

        // 根据提取器类型分发
        List<ExtractedNumber> results;
        switch (extractorType) {
            case "id_card":
                results = idCardExtractor.extract(text, getOptions(extractorConfig));
                break;
            case "bank_card":
                results = bankCardExtractor.extract(text, getOptions(extractorConfig));
                break;
            case "phone":
                results = phoneExtractor.extract(text, getOptions(extractorConfig));
                break;
            case "composite":
                results = extractComposite(text, extractorConfig);
                break;
            default:
                throw new IllegalArgumentException("未知的提取器类型: " + extractorType);
        }

        // 构建返回结果
        return buildResult(label.getExtractFields(), results, startTime);
    }

    /**
     * 复合提取（多种号码）
     */
    private List<ExtractedNumber> extractComposite(String text, JSONObject extractorConfig) {
        List<ExtractedNumber> allResults = new ArrayList<>();

        JSONArray extractors = extractorConfig.getJSONArray("extractors");
        for (int i = 0; i < extractors.size(); i++) {
            JSONObject extractorDef = extractors.getJSONObject(i);
            String extractorType = extractorDef.getString("extractorType");

            List<ExtractedNumber> partialResults;
            switch (extractorType) {
                case "id_card":
                    partialResults = idCardExtractor.extract(text, new HashMap<>());
                    break;
                case "bank_card":
                    partialResults = bankCardExtractor.extract(text, new HashMap<>());
                    break;
                case "phone":
                    partialResults = phoneExtractor.extract(text, new HashMap<>());
                    break;
                default:
                    continue;
            }

            // 添加字段标识
            String field = extractorDef.getString("field");
            for (ExtractedNumber result : partialResults) {
                result.setField(field);
            }

            allResults.addAll(partialResults);
        }

        return allResults;
    }
}
```
